Palindrome Checker

body { font-family: Arial, sans-serif; background: #f8f9fa; color: #333; text-align: center; padding-top: 50px; } .container { display: inline-block; padding: 20px; background: #ffffff; border-radius: 10px; box-shadow: 0 0 10px rgba(0,0,0,0.1); } .input-section { margin-bottom: 20px; } #text-input { padding: 10px; width: 60%; font-size: 16px; } #check-btn { padding: 10px 20px; font-size: 16px; margin-left: 10px; cursor: pointer; background: #007bff; border: none; color: #ffffff; border-radius: 5px; } #check-btn:hover { background: #0056b3; } #result { font-size: 18px; font-weight: bold; margin-top: 20px; } document.getElementById('check-btn').addEventListener('click', function() { const userInput = document.getElementById('text-input').value; const resultElement = document.getElementById('result'); // If no input provided, alert the user if(!userInput.trim()) { alert("Please input a value"); return; } // Normalize input: remove all non-alphanumeric chars and convert to lowercase const cleaned = userInput.toLowerCase().replace(/[\W_]/g, ''); // Check palindrome const isPalindrome = cleaned === cleaned.split('').reverse().join(''); if(isPalindrome) { resultElement.innerText = `${userInput} is a palindrome`; } else { resultElement.innerText = `${userInput} is not a palindrome`; } });
